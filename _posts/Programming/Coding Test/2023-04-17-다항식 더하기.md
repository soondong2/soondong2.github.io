## 프로그래머스 연속된 수의 합

- 난이도 : Level0

## 풀이
```python
def solution(polynomial):
    answer = ''
    num, x = 0, 0
    for p in polynomial.split(' + '):
        if 'x' not in p:
            num += int(p)
        elif p == 'x':
            x += 1
        elif len(p) > 1:
            x += int(p[:-1])
    
    if x == 0:
        answer = str(num)
    elif num == 0:
        if x == 1:
            answer = 'x'
        else:
            answer = str(x) + 'x'
    else:
        if x == 1:
            answer = 'x + ' + str(num)
        else:
            answer = str(x) + 'x + ' + str(num)
    return answer    
```
